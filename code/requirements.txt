# --- CORE ORCHESTRATION ---
# State-based orchestration framework for building agentic workflows.
# Essential for defining nodes, edges, and cycles in your application.
langgraph

# Pre-built components and utilities for common LangGraph patterns.
# Speeds up development by providing ready-made tools and logic.
langgraph-prebuilt

# SDK for interacting with LangGraph Cloud and remote deployments.
# Necessary if you plan to move from local notebooks to hosted graph APIs.
langgraph-sdk

# --- GOOGLE GEMINI INTEGRATION ---
# Official LangChain integration for Google Generative AI models (Gemini).
# Required to use ChatGoogleGenerativeAI within your LangGraph nodes.
langchain-google-genai

# --- DATA & STATE PERSISTENCE ---
# Core abstractions for LangChain components.
# Provides the base classes used by both LangGraph and Gemini integrations.
langchain-core>=1.2.28

# Community-maintained integrations for 3rd party tools and services.
# Useful for adding extra capabilities to your agents.
langchain-community

# Utilities for splitting long documents into manageable chunks.
# Critical for RAG patterns and managing Gemini's context window effectively.
langchain-text-splitters>=1.1.2

# SQLite-backed persistence layer for LangGraph checkpoints.
# Enables your agents to "remember" previous turns and handle interruptions.
langgraph-checkpoint-sqlite

# --- TOOLING & UTILITIES ---
# Tool for extracting structured data from unstructured text using LLMs.
# Used in the course for reliable schema-first tool calling.
trustcall

# Syntax highlighting for code in the terminal and notebooks.
# Makes the output of graph state and code snippets much more readable.
Pygments>=2.20.0

# Library for secure communication and data encryption.
# A low-level dependency required for secure API handshakes and authentication.
cryptography>=46.0.7

# --- OBSERVABILITY & SEARCH ---
# Client for LangSmith tracing, debugging, and evaluation.
# Vital for visualizing your graph execution and debugging agent logic.
langsmith>=0.7.31

# Search engine tool specifically optimized for LLM agents.
# Allows your Gemini-powered agent to fetch real-time web information.
langchain-tavily

# Wikipedia integration for LangChain tools.
# Provides a reliable source of factual data for agent research tasks.
wikipedia

# --- INFRASTRUCTURE ---
# Reads key-value pairs from a .env file and sets them as environment variables.
# Used to securely manage your Google and LangSmith API keys locally.
python-dotenv

# Local CLI for testing and managing LangGraph applications.
# Includes in-memory support for rapid local development.
langgraph-cli[inmem]

# Jupyter Notebook environment for interactive development.
# The primary interface used in the LangGraph Academy course.
notebook

# Asynchronous HTTP client for Python.
# Handles the underlying network requests for many LangChain tools.
aiohttp>=3.13.4
